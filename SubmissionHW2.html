%% Problem 4.5: Taylor Series Expansions
%Predict f(2.5) instead of f(3)
clear,clc
a=1;
x=2.5;
%f= @(x) 25*x^3-6*x^2+7*x-88;
f=25*a^3-6*a^2+7*a-88;
fp=75*a^2-12*a+7;
fpp=150*a-12;
fppp=150;
h=x-a;
T0=f;
T1=f+fp*h;
T2=f+fp*h+fpp*h^2/factorial(2);
T3=f+fp*h+fpp*h^2/factorial(2)+fppp*h^3/factorial(3);
Tr=25*x^3-6*x^2+7*x-88;
E0=(abs(T0-Tr)/Tr)*100;
E1=(abs(T1-Tr)/Tr)*100;
E2=(abs(T2-Tr)/Tr)*100;
E3=(abs(T3-Tr)/Tr)*100;
E=[E0 E1 E2 E3];
bar(E)
xlabel('Expansion Order')
ylabel('True Percent Relative Error')
title('Relative Error')
